 13def qm_thermo(atom_coord_path=None, atom_numbers=None, coords=None,
 14              vib_path=None, vibfreqs=None,
 15              ee_path=None, ee=None,
 16              T=298.15, P=101325,
 17              sclZPE=1.0, sclU=1.0, sclCv=1.0, sclS=1.0,

 19              read_ee_index=-1,
 20              E_list=None, g_list=None,
 21              ignore_trans_and_rot=False,
 22              c=None):
 23    """
 24    Calculates thermodynamic properties based on quantum mechanical data.
 25
 26    The function computes the translational, rotational, vibrational, and electronic contributions to thermodynamic
 27    quantities such as internal energy (U), enthalpy (H), Gibbs free energy (G), heat capacities (Cv, Cp), and entropy
 28    (S). It supports both direct input of atomic coordinates, vibrational frequencies, and electronic energies, or
 29    loading these from files. The function also allows for scaling of various contributions and correction methods
 30    for more accurate results.
 31
 32    Parameters:
 33    - atom_coord_path (str):
 34            Path to the file containing atomic coordinates.
 35    - atom_numbers (list[int]):
 36            List of atomic numbers.
 37    - coords (np.ndarray):
 38            Array of atomic coordinates.
 39    - vib_path (str):
 40            Path to the file containing vibrational frequencies.
 41    - vibfreqs (list[float]):
 42            List of vibrational frequencies in cm^-1.
 43    - ee_path (str):
 44            Path to the file containing electronic energies.
 45    - ee (float):
 46            Electronic energy in Hartrees.
 47    - T (float):
 48            Temperature in Kelvin.
 49    - P (float):
 50            Pressure in Pascals.
 51    - sclZPE (float):
 52            Scaling factor for zero-point energy.
 53    - sclU (float):
 54            Scaling factor for internal energy.
 55    - sclCv (float):
 56            Scaling factor for constant volume heat capacity.
 57    - sclS (float):
 58            Scaling factor for entropy.
 59    - U_Minenkov (bool):
 60            Flag to use Minenkov's method for internal energy correction.
 61    - S_Grimme (bool):
 62            Flag to use Grimme's method for entropy correction.
 63    - verbose (bool):
 64            Flag to print detailed output.
 65    - read_ee_index (int):
 66            Index to select the electronic energy value when multiple values are available. (When use ee_path rather than ee)
 67    - E_list (list[float]):
 68            List of electronic energy levels.
 69    - g_list (list[int]):
 70            List of degeneracies corresponding to the electronic energy levels.
 71    - ignore_trans_and_rot (bool):
 72            Flag to ignore translational and rotational contributions.
 73    - c (float):
 74            Concentration in mol/m^3 for calculating concentration-dependent Gibbs free energy.
 75
 76    Returns:
 77        (dict)
 78            A dictionary containing the calculated thermodynamic properties.
 79
 80    Raises:
 81        ValueError
 82            If neither `atom_coord_path` nor both `atom_numbers` and `coords` are provided.
 83    """
 84    # load data
 85    # 如果提供了atom_numbers和coords，则不需要从atom_coord_path读取
 86    if atom_numbers is None or coords is None:
 87        if atom_coord_path is None:
 88            raise ValueError("Either atom_coord_path or (atom_numbers and coords) must be provided.")
 89        atom_numbers, coords = read_atom_coord(atom_coord_path)
 90    atom_masses = np.array([atom_data[i][3] for i in atom_numbers])
 91    M = np.sum(atom_masses)
 92
 93    # 处理振动频率输入
 94    if vibfreqs is None:
 95        if vib_path is None:  # 如果未提供振动频率列表且没有指定路径，则使用原子坐标路径
 96            vib_path = atom_coord_path
 97        vibfreqs = read_vib(vib_path)
 98
 99    # 处理电子能量输入
100    if ee is None:
101        if ee_path is None:  # 如果未提供电子能量值且没有指定路径，则使用原子坐标路径
102            ee_path = atom_coord_path
103        ee = read_ee(ee_path, ee_index=read_ee_index)
104
105    if E_list is None:
106        E_list = [ee]
107    if g_list is None:
108        g_list = [1]
109
110    # calculate
111    # trans
112    q_t, U_t, H_t, Cv_t, Cp_t, S_t = contribution_trans(M=M, T=T, P=P)
113    if verbose:
114        print('======= Translation =======')
115        print(f'q_t: {q_t}')
116        print(f'U_t: {U_t}')
117        print(f'H_t: {H_t}')
118        print(f'Cv_t: {Cv_t}')
119        print(f'Cp_t: {Cp_t}')
120        print(f'S_t: {S_t}')
121    # rot
122    q_r, U_r, H_r, Cv_r, Cp_r, S_r = contribution_rot(atom_numbers=atom_numbers, coords=coords, T=T, convert_unit=True)
123    if verbose:
124        print('========= Rotation ========')
125        print(f'q_r: {q_r}')
126        print(f'U_r: {U_r}')
127        print(f'H_r: {H_r}')
128        print(f'Cv_r: {Cv_r}')
129        print(f'Cp_r: {Cp_r}')
130        print(f'S_r: {S_r}')
131    # vib
132    q_v_0, q_v_bot, U_v_0_T, H_v_0_T, U_v, H_v, Cv_v, Cp_v, S_v, zpe = contribution_vib(vibfreqs=vibfreqs, T=T, convert_unit=True,
133                                                                                        sclZPE=sclZPE, sclU=sclU, sclCv=sclCv, sclS=sclS,
134                                                                                        U_Minenkov=U_Minenkov, S_Grimme=S_Grimme)
135    if verbose:
136        print('======== Vibration ========')
137        print(f'q_v_bot: {q_v_bot}')
138        print(f'q_v_0: {q_v_0}')
139        print(f'zpe: {zpe}')
140        print(f'U_v_0_T: {U_v_0_T}')
141        print(f'H_v_0_T: {H_v_0_T}')
142        print(f'U_v: {U_v}')
143        print(f'H_v: {H_v}')
144        print(f'Cv_v: {Cv_v}')
145        print(f'Cp_v: {Cp_v}')
146        print(f'S_v: {S_v}')
147    # ele
148    q_e, U_e, H_e, Cv_e, Cp_e, S_e = contribution_ele(E_list=E_list, g_list=g_list, T=T, convert_unit=True)
149    if verbose:
150        print('======== Electron excitation ========')
151        print(f'q_e: {q_e}')
152        print(f'U_e: {U_e}')
153        print(f'H_e: {H_e}')
154        print(f'Cv_e: {Cv_e}')
155        print(f'Cp_e: {Cp_e}')
156        print(f'S_e: {S_e}')
157
158    # total
159    if ignore_trans_and_rot:
160        q_tot_v_0 = q(q_t=q_t, q_r=q_r, q_v=q_v_0, q_e=q_e, ignore_trans_and_rot=ignore_trans_and_rot)
161        q_tot_bot = q(q_t=q_t, q_r=q_r, q_v=q_v_bot, q_e=q_e, ignore_trans_and_rot=ignore_trans_and_rot)
162        Cv_tot = Cv_v + Cv_e
163        Cp_tot = Cp_v + Cp_e
164        S_tot = S_v + S_e
165        U_corr = U_v + U_e
166        H_corr = H_v + H_e
167        G_corr = H_corr - T * S_tot
168    else:
169        q_tot_v_0 = q(q_t=q_t, q_r=q_r, q_v=q_v_0, q_e=q_e)
170        q_tot_bot = q(q_t=q_t, q_r=q_r, q_v=q_v_bot, q_e=q_e)
171        Cv_tot = Cv_t + Cv_r + Cv_v + Cv_e
172        Cp_tot = Cp_t + Cp_r + Cp_v + Cp_e
173        S_tot = S_t + S_r + S_v + S_e
174        U_corr = U_t + U_r + U_v + U_e
175        H_corr = H_t + H_r + H_v + H_e
176        G_corr = H_corr - T * S_tot
177
178    zpe_plus_ee = ee * au2j_mol  + zpe
179    U = ee * au2j_mol + U_corr
180    H = ee * au2j_mol  + H_corr
181    G = ee * au2j_mol  + G_corr
182
183    if c is not None:
184        c_gas = P / (R * T)
185        delta_G_of_conc = R * T * np.log(c / c_gas)
186        G += delta_G_of_conc
187    if verbose:
188        print('========== Total ==========')
189        print(f'Ignore contribution of trans and rot? {ignore_trans_and_rot}')
190        print(f'q(V=0): {q_tot_v_0}')
191        print(f'q(bot): {q_tot_bot}')
192        print(f'Cv_tot: {Cv_tot} J/mol/K')
193        print(f'Cp_tot: {Cp_tot} J/mol/K')
194        print(f'S_tot: {S_tot} J/mol/K')
195        print(f'ZPE: {zpe} J/mol')
196        print(f'U_corr: {U_corr} J/mol')
197        print(f'H_corr: {H_corr} J/mol')
198        print(f'G_corr: {G_corr} J/mol')
199        print(f'EE: {ee * au2j_mol} J/mol {ee} a.u.')
200        print(f'Sum of electronic energy and ZPE, namely U/H/G at 0 K: {zpe_plus_ee} J/mol {zpe_plus_ee/au2j_mol} a.u.')
201        print(f'Sum of electronic energy and thermal correction to U: {U}  J/mol {U/au2j_mol} a.u.')
202        print(f'Sum of electronic energy and thermal correction to H: {H}  J/mol {H/au2j_mol} a.u.')
203        print(f'Sum of electronic energy and thermal correction to G: {G}  J/mol {G/au2j_mol} a.u.')
204        if c is not None:
205            print(f'delta-G of conc. change: {delta_G_of_conc}  J/mol {delta_G_of_conc / au2j_mol} a.u.')
206
207    if c is not None:
208        return {
209            'T/K': T, 'P/Pa': P,
210            'q_tot_v_0': q_tot_v_0, 'q_tot_bot': q_tot_bot,
211            'Cv/(J/mol/K)': Cv_tot, 'Cp/(J/mol/K)': Cp_tot, 'S/(J/mol/K)': S_tot,
212            'zpe/(J/mol)': zpe, 'U_corr/(J/mol)': U_corr, 'H_corr/(J/mol)': H_corr, 'G_corr/(J/mol)': G_corr,
213            'ee/(J/mol)': ee * au2j_mol, 'U/(J/mol)': U, 'H/(J/mol)': H, 'G/(J/mol)': G,
214            'delta_G_of_conc.(J/mol)': delta_G_of_conc,
215        }
216    else:
217        return {
218            'T/K': T, 'P/Pa': P,
219            'q_tot_v_0': q_tot_v_0, 'q_tot_bot': q_tot_bot,
220            'Cv/(J/mol/K)': Cv_tot, 'Cp/(J/mol/K)': Cp_tot, 'S/(J/mol/K)': S_tot,
221            'zpe/(J/mol)': zpe, 'U_corr/(J/mol)': U_corr, 'H_corr/(J/mol)': H_corr, 'G_corr/(J/mol)': G_corr,
222            'ee/(J/mol)': ee * au2j_mol, 'U/(J/mol)': U, 'H/(J/mol)': H, 'G/(J/mol)': G,
223        }

248    Parameters:
249    - atom_coord_path (str):
250            Path to the file containing atomic coordinates.
251    - atom_numbers (list[int]):
252            List of atomic numbers.

254            Array of atomic coordinates.
255    - vib_path (str):
256            Path to the file containing vibrational frequencies.
257    - vibfreqs (list[float]):
258            List of vibrational frequencies in cm^-1.
259    - ee_path (str):
260            Path to the file containing electronic energies.
261    - ee (float):
262            Electronic energy in Hartrees.

277    - S_Grimme (bool):
278            Flag to use Grimme's method for entropy correction.

282            Index to select the electronic energy value when multiple values are available. (When use ee_path rather than ee)
283    - E_list (list[float]):
284            List of electronic energy levels.
285    - g_list (list[int]):
286            List of degeneracies corresponding to the electronic energy levels.
287    - ignore_trans_and_rot (bool):
288            Flag to ignore translational and rotational contributions.
289    - c (float):
290            Concentration in mol/m^3 for calculating concentration-dependent Gibbs free energy.
291    - out_path: str, default 'QMthermoScan.xlsx'

314                               verbose=False, c=c
315                               )
316            results.append(result)
317    df = pd.DataFrame(results)
318    if out_path is not None:
319        df.to_excel(out_path, index=False)

323def qm_thermo_conformation_weighting(
324    U_list, H_list, G_list, S_list, Cv_list, Cp_list, T=298.15
325):
326    """
327    Calculates the thermodynamic properties of a system based on the conformational weighting
328    of its internal energy (U), enthalpy (H), Gibbs free energy (G), entropy (S), and heat capacities (Cv, Cp) at a given temperature.
329
330    Summary:
331    This function computes the weighted average of thermodynamic properties for a set of conformations.
332    It first converts any input lists to numpy arrays for efficient computation. The weighting is done
333    using the Gibbs free energies (G_list) and the provided or default temperature (T).
334    The function returns a dictionary containing the weights of each conformation and the calculated
335    thermodynamic properties: U, H, G, S, Cv, and Cp, along with the temperature used in the calculation.
336
337    Parameters:
338    - U_list (List[float] | np.ndarray): List or array of internal energies for each conformation in J/mol.
339    - H_list (List[float] | np.ndarray): List or array of enthalpies for each conformation in J/mol.
340    - G_list (List[float] | np.ndarray): List or array of Gibbs free energies for each conformation in J/mol.
341    - S_list (List[float] | np.ndarray): List or array of entropies for each conformation in J/(mol·K).
342    - Cv_list (List[float] | np.ndarray): List or array of constant volume heat capacities for each conformation in J/(mol·K).
343    - Cp_list (List[float] | np.ndarray): List or array of constant pressure heat capacities for each conformation in J/(mol·K).
344    - T (float, optional): Temperature in Kelvin. Defaults to 298.15 K.
345
346    Returns:
347        (dict): A dictionary with keys 'weight', 'U/(J/mol)', 'H/(J/mol)', 'G/(J/mol)', 'S/(J/mol)',
348              'Cv/(J/mol)', 'Cp/(J/mol)', and 'T/K' corresponding to the calculated values.
349
350    Raises:
351        ValueError: If any of the lists have different lengths.
352    """
353    if isinstance(U_list, list):
354        U_list = np.array(U_list)
355    if isinstance(H_list, list):
356        H_list = np.array(H_list)
357    if isinstance(Cv_list, list):
358        Cv_list = np.array(Cv_list)
359    if isinstance(Cp_list, list):
360        Cp_list = np.array(Cp_list)
361    if isinstance(S_list, list):
362        S_list = np.array(S_list)
363    if isinstance(G_list, list):
364        G_list = np.array(G_list)
365
366    p = calculate_conformation_weighting(G_list=G_list, T=T)
367    U = np.sum(U_list * p)
368    H = np.sum(H_list * p)
369    Cv = np.sum(Cv_list * p)
370    Cp = np.sum(Cp_list * p)
371    G = np.sum(G_list * p)
372    S = np.sum(S_list * p) - R * np.sum(p * np.log(p))
373    result = {
374        'weight': p,
375        'U/(J/mol)': U,
376        'H/(J/mol)': H,
377        'G/(J/mol)': G,
378        'S/(J/mol)': S,
379        'Cv/(J/mol)': Cv,
380        'Cp/(J/mol)': Cp,
381        'T/K': T
382    }
383    return result

386def calculate_conformation_weighting(G_list, T=298.15):
387    """
388    Calculate the conformation weighting for a list of free energy values.
389
390    This function computes the Boltzmann distribution of conformations based on their
391    free energies, which is essential in statistical mechanics and molecular simulations
392    to understand the relative probabilities of different states or conformations at a given temperature.
393
394    Parameters:
395    ----------
396    - G_list (List[float] or np.ndarray):
397        A list or numpy array containing the free energy values of each conformation. unit is eV.
398    T (float):
399        The temperature in Kelvin at which to evaluate the Boltzmann distribution. Default is 298.15 K.
400
401    Returns:
402    -------
403    (np.ndarray)
404        An array representing the probability of each conformation occurring, normalized such that
405        the sum of all probabilities equals one.
406
407    Raises:
408    ------
409    - ValueError
410        If G_list is not a list or numpy array, or if T is not a positive number.
411    - TypeError
412        If elements in G_list are not numeric.
413
414    Notes:
415    -----
416    - The function first converts G_list to a numpy array if it is provided as a list.
417    - It then calculates the difference between each free energy value and the minimum free energy in the list.
418    - Using these differences, it computes the Boltzmann factor for each conformation and normalizes them to get the probabilities.
419    - R (the gas constant) is assumed to be defined elsewhere in the code, typically with a value of 8.314 J/(mol·K).
420    """
421    if isinstance(G_list, list):
422        G_list = np.array(G_list)
423    delta_G = G_list - G_list.min()
424    p = np.exp(-delta_G / (R * T)) / np.sum(np.exp(-delta_G / (R * T)))
425    return p

458def contribution_rot(atom_numbers, coords, T, convert_unit=True):
459    """
460    Calculate the rotational contribution to thermodynamic properties for a molecule.
461
462    This function computes the partition function (q_r), internal energy (U_r), enthalpy (H_r),
463    constant volume heat capacity (Cv_r), constant pressure heat capacity (Cp_r), and entropy (S_r)
464    for rotational degrees of freedom. It handles both single-atom and multi-atom molecules,
465    accounting for linear and non-linear molecular structures, and applies unit conversion if required.
466
467    Parameters:
468    - atom_numbers (List[int]): List of atomic numbers representing the molecule.
469    - coords (List[List[float]]): 3D coordinates of the atoms in the molecule.
470    - T (float): Temperature at which to calculate the thermodynamic properties.
471    - convert_unit (bool, optional): Whether to convert moment of inertia units. Defaults to True.
472
473    Returns:
474    - Tuple[float, float, float, float, float, float]: A tuple containing q_r, U_r, H_r, Cv_r, Cp_r, S_r respectively.
475
476    Raises:
477    - ValueError: If the input lists for atom_numbers or coords are empty or of incorrect length.
478    - TypeError: If any of the inputs are not of the expected type.
479
480    Notes:
481    - The function uses specific formulas for linear and non-linear molecules, and adjusts
482      calculations based on the point group symmetry of the molecule.
483    - Unit conversion for the moment of inertia is applied if `convert_unit` is set to True.
484    - For a single-atom molecule, all rotational contributions are zero except for the partition function.
485    - This function relies on several helper functions (`get_I`, `check_linear`, `get_point_group`,
486      `get_rotational_symmetry_number`, and various `*_rot_*` functions) that must be defined elsewhere

The function computes partition function (q_e), internal energy (U_e), enthalpy (H_e), heat capacities at constant volume (Cv_e) and constant pressure (Cp_e), and entropy (S_e).
